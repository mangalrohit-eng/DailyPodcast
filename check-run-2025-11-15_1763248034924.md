.then(data => {
  console.log('=== ERRORS FOUND ===');
  console.log(data);

### 2. Check manifest for error details:
```javascript
fetch('/api/runs?runId=2025-11-15_1763248034924', {
  headers: {'Authorization': `Bearer ${localStorage.getItem('auth_token')}`}
})
.then(r => r.json())
.then(data => {
  console.log('=== RUN STATUS ===');
  console.log('Status:', data.summary?.status);
  console.log('Error:', data.summary?.error);
  console.log('Started:', data.summary?.started_at);
  console.log('Completed:', data.summary?.completed_at);
  console.log('\n=== MANIFEST ===');
  console.log(data.manifest);
});
```

### 3. Check agent logs:
```javascript
fetch('/api/external?action=logs&runId=2025-11-15_1763248034924', {
  headers: {'Authorization': `Bearer ${localStorage.getItem('auth_token')}`}
})
.then(r => r.json())
.then(data => {
  console.log('=== AGENT LOGS ===');
  console.log(`Found ${data.agent_count} agents that ran`);
  data.logs.forEach(log => {
    console.log(`\n${log.agent}:`);
    console.log('  Status:', log.status);
    console.log('  Duration:', log.duration_ms, 'ms');
    if (log.errors && log.errors.length > 0) {
      console.log('  ERRORS:', log.errors);
    }
  });
});
```
